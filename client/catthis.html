 29 let checkWins = () => {
 30   winner = 'none'   // Double checks that winner variable is set to right thing at first
 31   let drawCheck = 0
 32   for (let i = 0; i < winConditions.length; i++) {
 33     if (winConditions[i][0] !== 0) {
 34       if (winConditions[i][0] === winConditions[i][1] && winConditions[i][0] === winConditions[i][2] && winConditions[i][0] === 1) {
 35         winner = 'x'
 36       } else if (winConditions[i][0] === winConditions[i][1] && winConditions[i][0] === winConditions[i][2] && winConditions[i][0] === 2) {
 37         winner = 'o'
 38       } else if (winConditions[i].indexOf(0) === -1) {
 39         drawCheck += 1
 40       }
 41     }
 42   }
 43   if (drawCheck === 8) {
 44     winner = 'draw'
 45   }

169       } else if (turn === `${cyan}X`) {
170         boardState[Number(move)-1] = 1
171         alreadyTaken.push(Number(move))
